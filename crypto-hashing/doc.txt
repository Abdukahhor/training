ssl cert for server

    -x509 \
    -nodes \
    -newkey rsa:2048 \
    -keyout server.key \
    -out server.crt \
    -days 1000 \
    -subj "/C=TJ/ST=Dushanbe/L=Dushanbe/O=Alif Capital/OU=IT Department/CN=*"
